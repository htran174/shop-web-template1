  <script>
    function heroSlider() {
      return {
        // images (use your placeholders; change or add easily)
        slides: [
          { src: '/assets/images/img1.jpg',      alt: 'Salon nails — 1' },
          { src: '/assets/images/img2.jpg',      alt: 'Salon nails — 2' },
          { src: '/assets/images/img3.jpg',      alt: 'Salon nails — 3' },
          { src: '/assets/images/img4.jpg',      alt: 'Salon nails — 4' },
          { src: '/assets/images/shop.jpg',      alt: 'Salon nails — 5' },
        ],
        current: 0,
        intervalMs: 8000,       // ~6s per slide (luxury pace)
        _timer: null,
        init() {
          // Respect prefers-reduced-motion: disable auto-advance
          const reduce = window.matchMedia('(prefers-reduced-motion: reduce)').matches;
          if (!reduce) this._start();
        },
        _start() {
          this._timer = setInterval(() => this.next(), this.intervalMs);
        },
        pause() {
          if (this._timer) { clearInterval(this._timer); this._timer = null; }
        },
        resume() {
          if (!this._timer) this._start();
        },
        next() {
          this.current = (this.current + 1) % this.slides.length; // infinite loop
        },
        go(i) {
          this.current = i;
          // soft reset autoplay so it doesn’t immediately jump
          this.pause();
          this.resume();
        }
      }
    }
  </script>
